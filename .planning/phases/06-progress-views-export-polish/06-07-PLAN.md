**InSessionPRState (@Observable per project rules — PATTERNS §"No Analog Found" + UI-SPEC § Component Inventory row InSessionPRState):**
```
@Observable
public final class InSessionPRState {
    public var prTable: PRTable
    public var currentBanner: InSessionPRBannerContent?
    private var dismissTask: Task<Void, Never>?

    public init(prTable: PRTable = PRTable()) {
        self.prTable = prTable
    }

    /// Called from SessionLoggerView after a working SetEntry is committed.
    /// Mutates prTable in place when new PRs hit; updates currentBanner to drive the .safeAreaInset render.
    @MainActor
    public func observeSave(set: SetEntry) {
        let hits = PRDetector.check(set: set, against: &prTable)
        if !hits.isEmpty {
            let ordered = PRKind.allCases.filter { hits.contains($0) }   // canonical order weight→reps→volume→e1RM
            currentBanner = InSessionPRBannerContent(kinds: ordered, weight: set.weight, reps: set.reps)
            UINotificationFeedbackGenerator().notificationOccurred(.success)
            dismissTask?.cancel()
            dismissTask = Task { [weak self] in
                try? await Task.sleep(for: .seconds(5))
                await MainActor.run { self?.currentBanner = nil }
            }
        } else if currentBanner != nil {
            // Next save with no PRs dismisses prior banner (D-15)
            currentBanner = nil
            dismissTask?.cancel()
        }
    }

    @MainActor public func dismiss() {
        currentBanner = nil
        dismissTask?.cancel()
    }
}

public struct InSessionPRBannerContent: Sendable, Equatable {
    public let kinds: [PRKind]
    public let weight: Double
    public let reps: Int
}
```

**SessionFactory.seedPRTable (RESEARCH Pitfall 6 — per-exercise loop locked):**

The per-exercise loop is the LOCKED implementation path, NOT a fallback. Realistic routines hold 6–12 SessionExercises, so the loop runs 6–12 single-keyed fetches — well within budget. This deliberately avoids the SwiftData macro-compiler risk that comes with cross-relationship predicates (`entry.sessionExercise?.exercise?.id`), which the iOS 18 macro compiler may still reject in 2026. RESEARCH Pitfall 6 endorses this path as the recommended option for v1.

```
extension SessionFactory {
    /// Seeds the PR table at session start by issuing one keyed fetch per
    /// exerciseID in the current session, then composing the results via
    /// PRDetector.buildTable. Bounded by routine.exercises.count (~6–12).
    @MainActor
    public static func seedPRTable(for session: Session, context: ModelContext) async -> PRTable {
        let exerciseIDs: [UUID] = (session.exercises ?? []).compactMap { $0.exercise?.id }
        guard !exerciseIDs.isEmpty else { return PRTable() }

        var aggregated: [SetEntry] = []
        for exID in exerciseIDs {
            let targetID = exID   // local-let capture per Pitfall 3
            let descriptor = FetchDescriptor<SetEntry>(
                predicate: #Predicate<SetEntry> { entry in
                    entry.isComplete
                        && !entry.isWarmup
                        && entry.sessionExercise?.exercise?.id == targetID
                },
                sortBy: [SortDescriptor(\.completedAt)]
            )
            if let sets = try? context.fetch(descriptor) {
                aggregated.append(contentsOf: sets)
            }
        }
        return PRDetector.buildTable(from: aggregated)
    }
}
```
