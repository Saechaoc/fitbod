**Target signatures:**
```
public struct WeeklyTonnageBar: Sendable, Equatable {
    public let weekStart: Date          // ISO-8601 Monday at 00:00 local
    public let totalTonnage: Double     // sum(weight * reps) over working sets
    public let muscleBreakdown: [MuscleSlice]?   // nil when no muscle filter selected
}

public struct MuscleSlice: Sendable, Equatable {
    public let muscleID: UUID
    public let muscleName: String
    public let tonnage: Double
}

public enum TimeRangeChoice: String, CaseIterable, Sendable { case last8wk = "8wk"; case last26wk = "26wk"; case allTime = "all" }

public enum TonnageSliceMode: String, CaseIterable, Sendable { case timeRange; case blockPhase; case muscle }

public protocol MuscleVolumeProvider: Sendable {
    /// Returns the per-muscle tonnage contribution for one logged set.
    /// Phase 6 fallback (Unweighted): each primary muscle gets full set tonnage; secondary muscles get 0.
    /// Phase 5 will conform StimulusWeightedMuscleVolumeProvider that multiplies by ExerciseMuscleStimulus.weight.
    func contributions(for set: SetEntry, exercise: Exercise) -> [(muscleID: UUID, muscleName: String, tonnage: Double)]
}

public struct UnweightedMuscleVolumeProvider: MuscleVolumeProvider {
    public init()
}

public enum WeeklyTonnageAggregator {
    public static func aggregate(
        sets: [SetEntry],
        weekStartsMonday: Bool,
        selectedMuscleIDs: Set<UUID>,
        provider: MuscleVolumeProvider
    ) -> [WeeklyTonnageBar]
}
```

**Aggregation behavior:**
1. Filter input to working sets only (setType == .working && !isWarmup && isComplete && weight > 0 && reps > 0).
2. Bucket by week-start via `Calendar.current.dateComponents([.yearForWeekOfYear, .weekOfYear], from: set.completedAt)` then construct the week-start date. Respect `weekStartsMonday` by setting `var calendar = Calendar.current; calendar.firstWeekday = weekStartsMonday ? 2 : 1` per RESEARCH "Don't Hand-Roll — Time-zone math for week start".
3. For each week bucket compute totalTonnage = sum(weight × reps).
4. If `selectedMuscleIDs.isEmpty`, muscleBreakdown = nil (single-color bar mode per D-19).
5. Else, for each working set in the bucket call `provider.contributions(for:exercise:)`, filter to selectedMuscleIDs, sum by muscleID, emit MuscleSlice array sorted by tonnage desc.
6. Return [WeeklyTonnageBar] sorted by weekStart asc.

**UnweightedMuscleVolumeProvider implementation:** Reads `exercise.muscleStimuli` (or whatever the SchemaV2 relationship is — verify by reading `Exercise.swift` + `ExerciseMuscleStimulus.swift`); for each stimulus with `role == .primary` (or weight ≥ 1.0 — adapt to actual field names), emits one tuple `(muscleID, muscleName, set.weight * Double(set.reps))`. Secondary muscles get 0 contribution (CONTEXT D-20 fallback semantics).
