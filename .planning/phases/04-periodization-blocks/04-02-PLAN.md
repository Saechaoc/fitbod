      ```
      public func save(into context: ModelContext) throws {
          try context.transaction {
              // D-05 single-active invariant — deactivate any other active block first.
              if isActive {
                  let myID = self.id
                  let descriptor = FetchDescriptor<Block>(predicate: #Predicate<Block> { $0.isActive == true })
                  let others = try context.fetch(descriptor)
                  for other in others where other.id != myID {
                      other.isActive = false
                  }
              }
              // Fetch existing or create new (three-way merge — but BlockPhase children are cascade-managed so we can recreate the array wholesale).
              let block: Block
              if let myID = self.id {
                  let targetID = myID
                  let descriptor = FetchDescriptor<Block>(predicate: #Predicate<Block> { $0.id == targetID })
                  block = try context.fetch(descriptor).first ?? {
                      let new = Block()
                      context.insert(new)
                      return new
                  }()
              } else {
                  block = Block()
                  context.insert(block)
                  self.id = block.id
              }
              // Materialize fields.
              block.name = name.trimmingCharacters(in: .whitespacesAndNewlines)
              block.startDate = startDate
              block.endDate = computedEndDate
              block.isActive = isActive
              block.reviewedAt = reviewedAt
              // Recreate cascade phases (BlockPhase has no grandchildren — wholesale replace is safe + simpler than three-way merge per 04-PATTERNS line 431).
              for old in (block.phases ?? []) { context.delete(old) }
              block.phases = phases.enumerated().map { (idx, draftPhase) -> BlockPhase in
                  let phase = BlockPhase()
                  phase.orderIndex = idx
                  phase.nameRaw = draftPhase.kind.rawValue
                  phase.weeks = draftPhase.weeks
                  phase.volumeMultiplier = draftPhase.volumeMultiplier
                  phase.intensityMultiplier = draftPhase.intensityMultiplier
                  context.insert(phase)
                  return phase
              }
              try context.save()
          }
      }
      ```
