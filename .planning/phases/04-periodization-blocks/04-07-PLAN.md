    ```swift
    // calendar week resolves to a deload phase, cut the working SetEntry count
    // by the phase's volumeMultiplier (clamped to >= 1 per RESEARCH §Pitfall 7).
    let scaledSetCount: Int

       let phase = PeriodizationEngine.phase(for: block, on: date),

        scaledSetCount = re.targetSets
    }
    ```
